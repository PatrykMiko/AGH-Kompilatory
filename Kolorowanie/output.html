// CalcLang example

fun square(x) = x * x

fun power(base, exp) =
    if exp == 0
        return 1
    else
        return base * power(base, exp - 1)

fun average(a, b) = (a + b) / 2

let pi = 3.14159
let r  = 5
let area = pi * square(r)
